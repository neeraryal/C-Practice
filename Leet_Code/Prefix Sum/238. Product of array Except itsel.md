Given an integer array nums, return an array answer such that answer[i] is equal to the product of all the elements of nums except nums[i].

The product of any prefix or suffix of nums is guaranteed to fit in a 32-bit integer.

You must write an algorithm that runs in O(n) time and without using the division operation.

 

Example 1:

Input: nums = [1,2,3,4]
Output: [24,12,8,6]
Example 2:

Input: nums = [-1,1,0,-3,3]
Output: [0,0,9,0,0]

```c
int* productExceptSelf(int* arr, int n, int* returnSize) 
{

    int * right_product_prefix = calloc(n,sizeof(int));
    right_product_prefix[0]=1;
    for(int i=1 ;i<n ;i++)
    {
        right_product_prefix[i]=arr[i-1]*right_product_prefix[i-1];
    }
    int* res= calloc(n,sizeof(int));
    int left_product=1, res_index=n-1;
    res[n-1]=1;
    for(int i=n-1 ; i>=0 ;i--)
    {
        res[res_index--]=left_product*right_product_prefix[i];
        left_product*=arr[i];
    }
    *returnSize=n;
    return res;
    
}
```
