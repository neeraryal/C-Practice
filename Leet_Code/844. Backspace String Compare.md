Given two strings s and t, return true if they are equal when both are typed into empty text editors. '#' means a backspace character.

Note that after backspacing an empty text, the text will continue empty.

 

Example 1:

Input: s = "ab#c", t = "ad#c"
Output: true
Explanation: Both s and t become "ac".
Example 2:

Input: s = "ab##", t = "c#d#"
Output: true
Explanation: Both s and t become "".
Example 3:

Input: s = "a#c", t = "b"
Output: false
Explanation: s becomes "c" while t becomes "b".


```c

int get_valid_index(char * s, int index) //Will reutn the valid index after processign # and if not a avlid index the return -1
{
    int count=0;
    while(index >=0)
    {
        if(s[index]=='#')
        {
            count++;
        }
        else 
        {
            if (count>0)
            {
                count--;
            }
            else{
                return index;
            }
        }
        index--;
    }
    return -1;
}

{
    int s_i= strlen(s)-1;
```
