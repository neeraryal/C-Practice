Given two strings s and t of lengths m and n respectively, return the minimum window substring of s such that every character in t (including duplicates) is included in the window. If there is no such substring, return the empty string "".

The testcases will be generated such that the answer is unique.

 

Example 1:

Input: s = "ADOBECODEBANC", t = "ABC"
Output: "BANC"
Explanation: The minimum window substring "BANC" includes 'A', 'B', and 'C' from string t.
Example 2:

Input: s = "a", t = "a"
Output: "a"
Explanation: The entire string s is the minimum window.
Example 3:

Input: s = "a", t = "aa"
Output: ""
Explanation: Both 'a's from t must be included in the window.
Since the largest window of s only has one 'a', return empty string.

```c

char* minWindow(char* s, char* t) {
    if (!s || !t || *s == '\0' || *t == '\0') {
        char* empty = malloc(1);
        empty[0] = '\0';
        return empty;
    }
    
    int* freq_t = calloc(128, sizeof(int));
    int required = 0;
    
    // Count frequency of each character in t
    for(int i = 0; t[i] != '\0'; i++) {
        freq_t[(unsigned char)t[i]]++;
        required++; // Total characters we need to match
    }
    
    int* freq_window = calloc(128, sizeof(int));
    int matched = 0; // How many characters from t we've matched
    int left = 0, right = 0;
    int min_len = __INT_MAX__;
    int min_start = 0;
    
    while(s[right] != '\0') {
        char c = s[right];
        
        // Add character to window
        if(freq_t[(unsigned char)c] > 0) {
            freq_window[(unsigned char)c]++;
            // Only count if we haven't exceeded the requirement
            if(freq_window[(unsigned char)c] <= freq_t[(unsigned char)c]) {
                matched++;
            }
        }
        
        // Try to shrink window from left while it's valid
        while(matched == required) {
            // Update minimum window
            if(right - left + 1 < min_len) {
                min_len = right - left + 1;
                min_start = left;
            }
            
            // Remove leftmost character
            char left_c = s[left];
            if(freq_t[(unsigned char)left_c] > 0) {
                freq_window[(unsigned char)left_c]--;
                // Only decrease matched if we go below requirement
                if(freq_window[(unsigned char)left_c] < freq_t[(unsigned char)left_c]) {
                    matched--;
                }
            }
            left++;
        }
        
        right++;
    }
    
    free(freq_t);
    free(freq_window);
    
    // No valid window found
    if(min_len == __INT_MAX__) {
        char* empty = malloc(1);
        empty[0] = '\0';
        return empty;
    }
    
    // Build result
    char* res = malloc(sizeof(char) * (min_len + 1));
    for(int i = 0; i < min_len; i++) {
        res[i] = s[min_start + i];
    }
    res[min_len] = '\0';
    
    return res;
}
```
