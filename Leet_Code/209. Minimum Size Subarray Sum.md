Given an array of positive integers nums and a positive integer target, return the minimal length of a subarray whose sum is greater than or equal to target. If there is no such subarray, return 0 instead.


Example 1:

Input: target = 7, nums = [2,3,1,2,4,3]
Output: 2
Explanation: The subarray [4,3] has the minimal length under the problem constraint.
Example 2:

Input: target = 4, nums = [1,4,4]
Output: 1
Example 3:

Input: target = 11, nums = [1,1,1,1,1,1,1,1]
Output: 0

```c

int minSubArrayLen(int target, int* arr, int n) {
    
    int left= 0 , right=0 ;
    int sum=0, min_window=INT_MAX;
    while(right<n)
    {
        sum+=arr[right];
        right++;
        
        while(sum>=target) //Do it while so that even if window with right at n-1 should also be considered, else it will not enter loop 
        {
            int cuurent_window= right-left;
            min_window=min_window<cuurent_window?min_window:cuurent_window;
            sum-=arr[left];
            left++;
        }
    }
    return (min_window == INT_MAX) ? 0 : min_window;
}
```
